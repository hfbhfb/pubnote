	ep.Count(&countDep)
	fmt.Println(countDep)
	fmt.Println(countDep)
	op := ep.WithDeptNo("d003")
	fmt.Println(op)
	result,err := ep.GetByOption(op)
	if err != nil {
		fmt.Println(err.Error())
	}else{
		fmt.Println(result.DeptName)
		fmt.Println(result.DeptNo)
		fmt.Println(result.TableName())
	}
	ep.Begin()

}
